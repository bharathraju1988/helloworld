pipeline {
   agent any
   stages {
      stage('code checkout & Build') {
         steps {
            git 'https://github.com/bharathraju1988/helloworld.git'
            sh "mvn -Dmaven.test.failure.ignore=true clean package"
         }
        }
      stage('create docker image'){
           steps {
	      sh 'docker build -t helloworld .'
	      }
	      }
   }
}
